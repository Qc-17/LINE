02 — LINE-DOM · integrazione HTML
sorgente HTML + LINE 
<!-- Input HTML -->
<input id="nome" placeholder="Il tuo nome">
<button id="btn-saluta">Saluta</button>

<!-- Output HTML -->
<p id="output"></p>

<!-- Codice LINE -->
<script type="text/line"
        trigger="btn-saluta">
  <output>
    TALK Ciao, @nome!
    TALK Bentornato su LINE 3.0.
  </output>
</script>

<!-- Contatore con STAY -->
<button id="btn-inc">+</button>
<span id="count">0</span>

<script type="text/line"
        trigger="btn-inc"
        counter="count">
  <count>
    TALK @__N__
  </count>
</script>
