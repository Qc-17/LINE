02 LINE-DOM · integrazione HTML trigger · autorun · counter · 1 TALK per tag
sorgente HTML + LINE 
<!-- 1 TALK per tag, @nome iniettato in automatico -->
<input id="nome" placeholder="Il tuo nome">
<button id="btn-saluta">Saluta</button>
<p id="out-saluto"></p>

<script type="text/line"
        trigger="btn-saluta">
  <out-saluto>TALK Ciao, @nome! Benvenuto su LINE 3.0.</out-saluto>
</script>

<!-- Contatore con counter= -->
<button id="btn-inc">+</button>
<span id="count-display">0</span>

<script type="text/line"
        trigger="btn-inc"
        counter="count-display">
  <count-display>TALK @__N__</count-display>
</script>
